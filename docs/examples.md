## Работа с метаданными

### Инициализация конфигурации

```
# Инициализировать конфигурацию
metadata.init path="C:/Projects/MyConfig"

# Полная переиндексация
metadata.init path="C:/Projects/MyConfig" full_reindex=true
```

### Получение списка объектов

```
# Все справочники
metadata.list type="Catalog"

# Все документы
metadata.list type="Document"

# Все регистры накопления
metadata.list type="AccumulationRegister"
```

### Получение информации об объекте

```
# Полная информация о справочнике
metadata.get type="Catalog" name="Номенклатура"

# Результат:
{
  "name": "Номенклатура",
  "synonym": "Номенклатура",
  "uuid": "...",
  "attributes": [...],
  "tabular_sections": [...],
  "forms": [...],
  "templates": [...]
}
```

### Поиск объектов

```
# Поиск по имени
metadata.search query="Товар"

# Поиск с фильтром по типу
metadata.search query="Продажа" type="Document"

# Поиск по синониму
metadata.search query="Реализация" type="Document"
```

### Реквизиты объекта

```
# Получить реквизиты справочника
metadata.attributes type="Catalog" name="Номенклатура"

# Получить реквизиты табличной части
metadata.attributes type="Document" name="ПоступлениеТоваров" tabular_section="Товары"
```

### Дерево подсистем

```
# Корневые подсистемы
metadata.tree

# Подсистемы внутри "Продажи"
metadata.tree parent="Продажи"
```

### Связи объекта

```
# Где используется справочник
metadata.references type="Catalog" name="Номенклатура"

# Результат:
{
  "used_in_attributes": [
    {"object": "Документ.ПоступлениеТоваров.Товары", "attribute": "Номенклатура"},
    {"object": "РегистрНакопления.ОстаткиТоваров", "attribute": "Номенклатура"}
  ],
  "used_in_registers": [...]
}
```

## Анализ кода

### Получение кода модуля

```
# Модуль объекта документа
code.module type="Document" name="РеализацияТоваров" module_type="ObjectModule"

# Модуль менеджера
code.module type="Catalog" name="Номенклатура" module_type="ManagerModule"

# Модуль формы
code.module type="Document" name="РеализацияТоваров" module_type="FormModule" form="ФормаДокумента"
```

### Получение процедуры

```
# Конкретная процедура
code.procedure
  type="CommonModule"
  name="ОбщегоНазначения"
  procedure="ПолучитьЗначениеРеквизита"

# Результат:
{
  "name": "ПолучитьЗначениеРеквизита",
  "is_function": true,
  "is_export": true,
  "parameters": ["Объект", "ИмяРеквизита"],
  "code": "..."
}
```

### Поиск определения

```
# Найти где определена процедура
code.resolve
  identifier="ПолучитьЗначениеРеквизита"
  context_type="CommonModule"
  context_name="ОбщегоНазначенияКлиент"
```

### Поиск использований

```
# Где используется процедура
code.usages
  type="CommonModule"
  name="ОбщегоНазначения"
  procedure="ПолучитьЗначениеРеквизита"

# Результат:
{
  "usages": [
    {"module": "Документ.РеализацияТоваров.МодульОбъекта", "line": 45},
    {"module": "Документ.ПоступлениеТоваров.МодульОбъекта", "line": 32}
  ]
}
```

### Граф зависимостей

```
# Зависимости модуля
code.dependencies type="Document" name="РеализацияТоваров"

# Результат:
{
  "calls": ["ОбщегоНазначения.ПолучитьЗначение", "ПечатьДокументов.Печать"],
  "metadata_refs": ["Справочник.Номенклатура", "РегистрНакопления.ОстаткиТоваров"],
  "called_by": [...]
}
```

### Граф вызовов

```
# Граф вызовов процедур в модуле
code.callgraph type="Document" name="РеализацияТоваров"

# С ограничением глубины
code.callgraph type="Document" name="РеализацияТоваров" depth=2
```

### Расширенный анализ

```
# Полный анализ модуля
code.analyze type="Document" name="РеализацияТоваров"

# Результат:
{
  "procedures": [...],
  "functions": [...],
  "regions": [...],
  "variables": [...],
  "method_calls": [...],
  "metadata_accesses": [...],
  "queries": [...],
  "complexity": {...}
}
```

### Анализ сложности

```
# Цикломатическая сложность
code.complexity type="CommonModule" name="МодульПроведения"

# Результат:
{
  "total_complexity": 45,
  "procedures": [
    {"name": "ОбработкаПроведения", "complexity": 12},
    {"name": "ПроверитьОстатки", "complexity": 8}
  ]
}
```

### Валидация и линтинг

```
# Проверка синтаксиса
code.validate type="Document" name="РеализацияТоваров"

# Статический анализ (требует BSL LS)
code.lint type="Document" name="РеализацияТоваров"

# Форматирование
code.format type="CommonModule" name="ОбщегоНазначения"
```

## Генерация кода

### Генерация запросов

```
# Простой запрос выборки
generate.query
  template="select_all"
  table="Справочник.Номенклатура"

# Запрос с отбором
generate.query
  template="select_with_filter"
  table="Справочник.Номенклатура"
  filter_field="Родитель"
  filter_value="&Группа"

# Запрос с соединением
generate.query
  template="select_with_join"
  main_table="Документ.РеализацияТоваров"
  join_table="Справочник.Номенклатура"
  join_field="Номенклатура"
```

### Генерация обработчиков событий

```
# Обработчик ПередЗаписью
generate.handler
  template="before_write"
  object_type="Document"
  object_name="РеализацияТоваров"

# Обработчик ОбработкаПроведения
generate.handler
  template="posting"
  object_type="Document"
  object_name="РеализацияТоваров"

# Обработчик ОбработкаЗаполнения
generate.handler
  template="filling"
  object_type="Document"
  object_name="СчетНаОплату"
  basis="Документ.ЗаказКлиента"
```

### Генерация печатных форм

```
# Печатная форма документа
generate.print
  template="document_print_form"
  document="Счет"
  areas=["Шапка", "Товары", "Подвал"]

# С параметрами
generate.print
  template="document_with_table"
  document="ТоварнаяНакладная"
  table_section="Товары"
```

### Генерация движений

```
# Движения расхода
generate.movement
  template="expense"
  register="РегистрНакопления.ОстаткиТоваров"
  document="РеализацияТоваров"
  table_section="Товары"

# Движения прихода
generate.movement
  template="receipt"
  register="РегистрНакопления.ОстаткиТоваров"
  document="ПоступлениеТоваров"
  table_section="Товары"
```

### Генерация API методов

```
# Метод получения списка
generate.api
  template="get_list"
  object="Справочник.Номенклатура"

# Метод создания объекта
generate.api
  template="create"
  object="Документ.ЗаказКлиента"
```

## Работа с запросами

### Разбор запроса

```
query.parse text="
ВЫБРАТЬ
    Номенклатура.Наименование,
    Номенклатура.Артикул
ИЗ
    Справочник.Номенклатура КАК Номенклатура
ГДЕ
    Номенклатура.Родитель = &Группа
"

# Результат:
{
  "select_fields": ["Номенклатура.Наименование", "Номенклатура.Артикул"],
  "tables": [{"name": "Справочник.Номенклатура", "alias": "Номенклатура"}],
  "conditions": ["Номенклатура.Родитель = &Группа"],
  "parameters": ["Группа"]
}
```

### Валидация запроса

```
query.validate text="..."

# Проверяет:
# - Существование таблиц метаданных
# - Существование полей
# - Корректность типов
```

### Оптимизация запроса

```
query.optimize text="..."

# Результат:
{
  "suggestions": [
    {"type": "index", "message": "Добавьте индекс на поле Родитель"},
    {"type": "join", "message": "Используйте ЛЕВОЕ СОЕДИНЕНИЕ вместо вложенного запроса"}
  ]
}
```

### Объяснение запроса

```
query.explain text="..."

# Результат на человеческом языке:
"Запрос выбирает наименование и артикул из справочника Номенклатура,
отфильтровывая по указанной группе. Используется 1 таблица,
1 параметр запроса."
```

## Работа с макетами

### Получение структуры макета

```
template.get
  type="Document"
  name="СчетНаОплату"
  template="ПечатнаяФорма"

# Результат:
{
  "name": "ПечатнаяФорма",
  "areas": ["Шапка", "СтрокаТовара", "Итого", "Подвал"],
  "parameters": ["Организация", "Контрагент", "Сумма"]
}
```

### Параметры макета

```
template.parameters
  type="Document"
  name="СчетНаОплату"
  template="ПечатнаяФорма"

# Результат:
[
  {"name": "Организация", "type": "string"},
  {"name": "Номер", "type": "string"},
  {"name": "Дата", "type": "date"},
  {"name": "СуммаИтого", "type": "number"}
]
```

### Генерация кода заполнения

```
template.generate_fill_code
  type="Document"
  name="СчетНаОплату"
  template="ПечатнаяФорма"
  language="ru"

# Результат - готовый код заполнения макета:
Макет = ПолучитьМакет("ПечатнаяФорма");
ОбластьШапка = Макет.ПолучитьОбласть("Шапка");
ОбластьШапка.Параметры.Организация = Организация;
...
```

## Skills

### /1c-query

```
/1c-query Справочник.Номенклатура с отбором по Группе и сортировкой по Наименованию
```

Генерирует:
```bsl
Запрос = Новый Запрос;
Запрос.Текст = "
ВЫБРАТЬ
    Номенклатура.Ссылка,
    Номенклатура.Наименование
ИЗ
    Справочник.Номенклатура КАК Номенклатура
ГДЕ
    Номенклатура.Родитель = &Группа
УПОРЯДОЧИТЬ ПО
    Номенклатура.Наименование";

Запрос.УстановитьПараметр("Группа", Группа);
Результат = Запрос.Выполнить().Выгрузить();
```

### /1c-handler

```
/1c-handler ОбработкаПроведения для документа РеализацияТоваров с движениями по ОстаткиТоваров
```

### /1c-print

```
/1c-print для документа СчетНаОплату с реквизитами шапки и таблицей товаров
```

### /1c-movement

```
/1c-movement расход для документа РеализацияТоваров по регистру ОстаткиТоваров
```

### /1c-explain

```
/1c-explain
Для Каждого Строка Из Товары Цикл
    Движение = Движения.ОстаткиТоваров.Добавить();
    Движение.ВидДвижения = ВидДвиженияНакопления.Расход;
    Движение.Период = Дата;
    Движение.Номенклатура = Строка.Номенклатура;
    Движение.Склад = Склад;
    Движение.Количество = Строка.Количество;
КонецЦикла;
```

Объясняет код на человеческом языке.
